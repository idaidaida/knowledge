<script>
  (function(){
    // Kebab menu handlers
    const menuBtn = document.getElementById('postMenuBtn');
    const menu = document.getElementById('postMenu');
    function hideMenu(){ if(menu){ menu.classList.add('hidden'); menuBtn?.setAttribute('aria-expanded','false'); } }
    if (menuBtn && menu) {
      menuBtn.addEventListener('click', (e)=>{
        e.stopPropagation();
        const shown = !menu.classList.contains('hidden');
        if (shown) hideMenu(); else { menu.classList.remove('hidden'); menuBtn.setAttribute('aria-expanded','true'); }
      });
      document.addEventListener('click', (e)=>{
        if (!menu.classList.contains('hidden')) {
          const within = menu.contains(e.target) || menuBtn.contains(e.target);
          if (!within) hideMenu();
        }
      });
      document.addEventListener('keydown', (e)=>{ if(e.key === 'Escape') hideMenu(); });
    }

    // Notification dropdown
    const nbtn = document.getElementById('notifyBtn');
    const nmenu = document.getElementById('notifyMenu');
    function hideN(){ if(nmenu){ nmenu.classList.add('hidden'); nbtn?.setAttribute('aria-expanded','false'); } }
    if (nbtn && nmenu) {
      nbtn.addEventListener('click', (e)=>{
        e.stopPropagation();
        const shown = !nmenu.classList.contains('hidden');
        if (shown) hideN(); else { nmenu.classList.remove('hidden'); nbtn.setAttribute('aria-expanded','true'); }
      });
      document.addEventListener('click', (e)=>{
        if (!nmenu.classList.contains('hidden')) {
          const within = nmenu.contains(e.target) || nbtn.contains(e.target);
          if (!within) hideN();
        }
      });
      document.addEventListener('keydown', (e)=>{ if(e.key === 'Escape') hideN(); });
    }

    // Comment modal handlers
    const openBtn = document.getElementById('openCommentModal');
    const modal = document.getElementById('commentModal');
    if (!openBtn || !modal) return;
    const closeBtn = document.getElementById('closeCommentModal');
    const cancelBtn = document.getElementById('cancelCommentModal');
    const backdrop = document.getElementById('commentModalBackdrop');
    function open(){ modal.classList.remove('hidden'); modal.setAttribute('aria-hidden','false'); }
    function close(){ modal.classList.add('hidden'); modal.setAttribute('aria-hidden','true'); }
    openBtn.addEventListener('click', open);
    if (closeBtn) closeBtn.addEventListener('click', close);
    if (cancelBtn) cancelBtn.addEventListener('click', close);
    if (backdrop) backdrop.addEventListener('click', close);
    document.addEventListener('keydown', (e)=>{ if(e.key === 'Escape') close(); });
  })();
</script>
